<script>
(function () {
  const BACKEND_URL = "https://YOUR-BACKEND-DOMAIN.com";
  const launcher = document.getElementById("ay-chat-launcher");
  const box = document.getElementById("ay-chat-box");
  const messages = document.getElementById("ay-chat-messages");
  const input = document.getElementById("ay-chat-input");
  const sendBtn = document.getElementById("ay-chat-send");

  const transcript = [];

  function addMessage(text, type) {
    const el = document.createElement("div");
    el.className = "ay-msg " + (type === "user" ? "ay-user" : "ay-bot");
    el.textContent = text;
    messages.appendChild(el);
    messages.scrollTop = messages.scrollHeight;
  }

  async function sendLead() {
    try {
      await fetch(BACKEND_URL + "/api/lead", {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify({
          transcript,
          page_url: window.location.href,
          source: "squarespace_chatbot"
        })
      });
    } catch (err) {
      console.error("Lead send failed", err);
    }
  }

  async function sendMessage() {
    const text = input.value.trim();
    if (!text) return;

    addMessage(text, "user");
    transcript.push({ role: "user", content: text });
    input.value = "";

    try {
      const response = await fetch(BACKEND_URL + "/api/chat", {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify({
          message: text,
          history: transcript
        })
      });

      const data = await response.json();
      addMessage(data.reply, "bot");
      transcript.push({ role: "assistant", content: data.reply });

      const lower = (data.reply || "").toLowerCase();
      const userText = text.toLowerCase();

      if (
        userText.includes("my name is") ||
        userText.includes("my phone") ||
        userText.includes("call me") ||
        userText.includes("@") ||
        userText.match(/\d{3}[\s.-]?\d{3}[\s.-]?\d{4}/)
      ) {
        sendLead();
      }

      if (
        lower.includes("book") ||
        lower.includes("call") ||
        lower.includes("text us") ||
        lower.includes("text photos")
      ) {
        sendLead();
      }
    } catch (err) {
      addMessage("Sorry, something went wrong. Please call, text, or book online and our office will help you.", "bot");
      console.error(err);
    }
  }

  launcher.addEventListener("click", function () {
    const isOpen = box.style.display === "flex";
    box.style.display = isOpen ? "none" : "flex";

    if (!isOpen && messages.children.length === 0) {
      const opening = "Hi, this is A&Y Appliances. What appliance needs service today?";
      addMessage(opening, "bot");
      transcript.push({ role: "assistant", content: opening });
    }
  });

  sendBtn.addEventListener("click", sendMessage);

  input.addEventListener("keydown", function (e) {
    if (e.key === "Enter") {
      sendMessage();
    }
  });
})();
</script>
